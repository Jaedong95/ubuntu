$ sudo du -sh /home/jaedong/*   # 디렉토리의 파일 별 용량 조회 

2. zombie 쓰레드 죽이기 
$ sudo kill -9 [PID]   # 일반
$ sudo kill - 15 [PID]   # 강제종료

* defunct가 뜨면서 확실히 종료되지 않는 경우
$ ps -ef | grep defunct   # 부모 프로세스 찾기, UID  PID  PPID
$ sudo kill -9 [PPID]
